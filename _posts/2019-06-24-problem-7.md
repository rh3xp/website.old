#include <iostream>
#include <math.h>

using namespace std;

int main()
{
    
    int n = 0;
    int i;
    long unsigned int answer = 1;
    int modulo = pow(10, 9) + 7;
    
    cin >> n;
    
    int a[n];


    for(i=0; i<n; i++)
        cin >> a[i];
    
    for(i=0; i<n; i++)
    {
        answer = (answer * a[i]) % modulo;
    }
    
    cout << answer;
    
    return 0;
}
