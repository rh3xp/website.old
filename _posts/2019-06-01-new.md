#include <iostream>
using namespace std;

int main() {
    
    int n, i, j, isPrime;
    cin >> n;
    for(i = 2; i <= n; i++)
    {
        isPrime = 0;
        // Check whether i is prime or not
        for(j = 2; j <= i/2; j++){
             // Check If any number between 2 to i/2 divides I 
             // completely If yes the i cannot be prime number
             if(i % j == 0){
                 isPrime = 1;
                 break;
             }
        }
           
        if(isPrime==0 && n!= 1)
            cout << i << " ";
    }
    return 0;
}
