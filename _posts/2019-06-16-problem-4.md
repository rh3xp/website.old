``` c++
#include<string>
#include<cctype>
#include<iostream>
using namespace std;

char flip_letters(char c)
{
	if(isupper(c))
		return tolower(c);
	else
		return toupper(c);
}

int main()
{
	string str, new_str="";
	char x;
	getline(cin, str);
	for(int c=0; c < str.length(); c++)
	{
		x = flip_letters(str[c]);
		new_str += x;
	}
	cout << new_str;
	
	return 0;
}
```
