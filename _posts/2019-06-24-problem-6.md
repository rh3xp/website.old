The main challenge was taking multiple inputs on the same line. Ater that everything was simple.


``` c++
#include<iostream>

using namespace std;

int main()
{
    int l, r, k, count;
    cin >> l >> r >> k; 
    if ( (l > r) || (l > 1000) || (l < 1) || (r > 1000) || (r < 1) || (k > 1000) || (k < 1) )
        return -1;
    
    for(l; l<=r; l++)
    {
        if(l % k == 0)
            count ++;
    }
    
    cout << count;

    return 0;
}
```
